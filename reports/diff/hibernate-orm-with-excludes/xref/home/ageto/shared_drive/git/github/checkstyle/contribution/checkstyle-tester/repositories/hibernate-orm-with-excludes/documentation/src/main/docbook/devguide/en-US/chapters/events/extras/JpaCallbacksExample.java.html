1   @Entity
2   @EntityListeners( LastUpdateListener.class )
3   public class Cat {
4       @Id private Integer id;
5       private String name;
6       private Calendar dateOfBirth;
7       @Transient private int age;
8       private Date lastUpdate;
9       //getters and setters
10  
11      /**
12       * Set my transient property at load time based on a calculation,
13       * note that a native Hibernate formula mapping is better for this purpose.
14       */
15      @PostLoad
16      public void calculateAge() {
17          Calendar birth = new GregorianCalendar();
18          birth.setTime(dateOfBirth);
19          Calendar now = new GregorianCalendar();
20          now.setTime( new Date() );
21          int adjust = 0;
22          if ( now.get(Calendar.DAY_OF_YEAR) - birth.get(Calendar.DAY_OF_YEAR) &lt; 0) {
23              adjust = -1;
24          }
25          age = now.get(Calendar.YEAR) - birth.get(Calendar.YEAR) + adjust;
26      }
27  }
28  
29  public class LastUpdateListener {
30      /**
31       * automatic property set before any database persistence
32       */
33      @PreUpdate
34      @PrePersist
35      public void setLastUpdate(Cat o) {
36          o.setLastUpdate( new Date() );
37      }
38  }
